## React Usage

```jsx
import AUtags from '@gov.au/tags';

<AUtags tags={[
  {
    link: 'link/to/tag1',
    text: 'tag1',
  },
  {
    link: 'link/to/tag2',
    text: 'tag2',
  },
  {
    link: 'link/to/tag3',
    text: 'tag3',
    li: {
      className: 'li-wrapping-class',
    },
  },
]} />
```
